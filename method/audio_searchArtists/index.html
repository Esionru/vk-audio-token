audio.searchArtists

Search artists by name

Parameters:

NameValue
qText to search for
count (optional)Maximum number of artists to return
offset (optional)Offset to skip that number of artists

Example (Kate):

$query = "Justin Bieber";


curl_setopt(
    $ch,
    CURLOPT_URL,
    "https://api.vk.com/method/audio.searchArtists?access_token=".TOKEN."&q=".
    urlencode($query)."&count=3&v=5.95"
);
<?php


include __DIR__.'/../../autoloader.php';


use Vodka2\VKAudioToken\SupportedClients;


//Token obtained by example_microg.php script
define('TOKEN', $argv[1]);
define('USER_AGENT', SupportedClients::Kate()->getUserAgent());
$ch = curl_init();


curl_setopt($ch,CURLOPT_HTTPHEADER, array('User-Agent: '.USER_AGENT));
curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);


$query = "Justin Bieber";


curl_setopt(
    $ch,
    CURLOPT_URL,
    "https://api.vk.com/method/audio.searchArtists?access_token=".TOKEN."&q=".
    urlencode($query)."&count=3&v=5.95"
);


echo json_encode(json_decode(curl_exec($ch)), JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE)."\n\n";
